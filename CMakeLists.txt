kde4_header()


add_subdirectory( pics )



set(libkimagemapeditorcommon_SRC
    qextfileinfo.cpp 
    areacreator.cpp 
    kimearea.cpp 
    kimecommands.cpp 
    kimedialogs.cpp 
    kimagemapeditor.cpp 
    drawzone.cpp 
    arealistview.cpp 
    imageslistview.cpp 
    mapslistview.cpp
	)


########### next target ###############

set(kimagemapeditor_SRCS main.cpp kimeshell.cpp ${libkimagemapeditorcommon_SRC})

kde4_automoc(${kimagemapeditor_SRCS})

kde4_add_executable(kimagemapeditor ${kimagemapeditor_SRCS})

target_link_libraries(kimagemapeditor  ${KDE4_KDECORE_LIBS} ${KDE4_KPARTS_LIBS} ${KDE4_KHTML_LIBS} )

install_targets(/bin kimagemapeditor )


########### next target ###############

set(kimagemapeditor_PART_SRCS dummy.cpp ${libkimagemapeditorcommon_SRC})

kde4_automoc(${kimagemapeditor_PART_SRCS})

file(WRITE dummy.cpp "//autogenerated file by cmake\n")
kde4_add_plugin(kimagemapeditor WITH_PREFIX ${kimagemapeditor_PART_SRCS})

kde4_install_libtool_file( ${PLUGIN_INSTALL_DIR} kimagemapeditor )

target_link_libraries(kimagemapeditor  ${KDE4_KDECORE_LIBS} )

install_targets(${PLUGIN_INSTALL_DIR} kimagemapeditor )


########### next target ###############


########### install files ###############

install_files( ${XDG_APPS_DIR} FILES  kimagemapeditor.desktop )
install_files( ${SERVICES_INSTALL_DIR} FILES  kimagemapeditorpart.desktop )
install_files( ${DATA_INSTALL_DIR}/kimagemapeditor FILES  kimagemapeditorui.rc            kimagemapeditorpartui.rc )

kde4_footer()
